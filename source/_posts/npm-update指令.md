date: 2021-08-24 12:05:19

```bash
>> npm install -g npm-check     //安装npm-check
>> npm-check                    //查看系统插件是否需要升级
>> npm install -g npm-upgrade   //安装npm-upgrade
>> npm-upgrade        //更新package.json
>> npm update -g      //更新全局插件
>> npm update --save  //更新系统插件

// mac instsall hexo
brew install hexo
brew update

//mac update hexo
sudo npm-check -u
```
